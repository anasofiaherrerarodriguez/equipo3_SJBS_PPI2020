USE bvv3wriballi15fe5tut;

CREATE TABLE USUARIO(
DOCUMENTO NUMERIC PRIMARY KEY NOT NULL,
NOMBRE VARCHAR(50) NOT NULL,
APELLIDOVARCHAR(50) NOT NULL,
CORREO VARCHAR(50) NOT NULL
);

CREATE TABLE FASE(
CODIGO INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
NOMBRE VARCHAR(50) 
);

CREATE TABLE USUARIO_FASE(
DOCUMENTO_USUARIO NUMERIC NOT NULL,
CODIGO_FASE INT NOT NULL,
FOREIGN KEY(DOCUMENTO_USUARIO) REFERENCES USUARIO(DOCUMENTO),
FOREIGN KEY(CODIGO_FASE) REFERENCES FASE(CODIGO)
);


CREATE TABLE META(
CODIGO INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
NOMBRE VARCHAR(50) NOT NULL,
VALOR DECIMAL NOT NULL,
DOCUMENTO_USUARIO NUMERIC NOT NULL,
FOREIGN KEY(DOCUMENTO_USUARIO) REFERENCES USUARIO(DOCUMENTO)
);
